CREATE DATABASE av1_db


CREATE TABLE alunos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome varchar(100),
    email varchar(100),
    idade int (100),
    contato int (100),
    endereco varchar (100)
);

CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome varchar(100),
    email varchar(100),
    senha varchar(255)
);
